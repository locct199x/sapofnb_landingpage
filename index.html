    <script>

        lucide.createIcons();


        const apiKey = "";

        const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/imagen-3.0-generate-002:predict?key=${apiKey}`;

        

        // Cấu hình exponential backoff

        const MAX_RETRIES = 5;

        const INITIAL_DELAY = 1000; // 1 second


        /**

         * Hàm fetch với Exponential Backoff

         */

        async function exponentialBackoffFetch(url, options, maxRetries = MAX_RETRIES) {

            let delay = INITIAL_DELAY;

            for (let i = 0; i < maxRetries; i++) {

                try {

                    const response = await fetch(url, options);

                    if (response.status === 429 || response.status >= 500) {

                        // Retry on Rate Limit (429) or Server Errors (5xx)

                        throw new Error(`Server error or rate limit: ${response.status}`);

                    }

                    return response;

                } catch (error) {

                    if (i === maxRetries - 1) {

                        throw error; // Re-throw if last attempt failed

                    }

                    await new Promise(resolve => setTimeout(resolve, delay));

                    delay *= 2; // Double the delay

                }

            }

        }



        /**

         * Hàm gọi API Imagen để tạo ảnh

         * @param {string} prompt - Mô tả hình ảnh

         * @param {string} imageElementId - ID của thẻ <img> để hiển thị

         * @param {string} loadingElementId - ID của thẻ loading indicator

         */

        async function generateImage(prompt, imageElementId, loadingElementId) {

            const imageElement = document.getElementById(imageElementId);

            const loadingElement = document.getElementById(loadingElementId);

            const containerElement = imageElement.parentElement;


            if (!imageElement || !loadingElement) return;

            

            loadingElement.classList.remove('hidden');

            imageElement.classList.add('hidden');

            

            // Xóa nội dung nền của container để hiển thị spinner

            containerElement.style.backgroundImage = 'none';


            const payload = {

                instances: [{ prompt: prompt }],

                parameters: { "sampleCount": 1 }

            };


            try {

                const response = await exponentialBackoffFetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

                

                const result = await response.json();

                

                if (result.predictions && result.predictions.length > 0 && result.predictions[0].bytesBase64Encoded) {

                    const base64Data = result.predictions[0].bytesBase64Encoded;

                    const imageUrl = `data:image/png;base64,${base64Data}`;

                    

                    imageElement.src = imageUrl;

                    imageElement.classList.remove('hidden');

                } else {

                    console.error("Lỗi: Không nhận được dữ liệu ảnh từ API.");

                    containerElement.textContent = "Lỗi tải ảnh.";

                }


            } catch (error) {

                console.error("Lỗi khi tạo ảnh:", error);

                containerElement.textContent = "Lỗi tải ảnh (Vui lòng thử lại sau).";

            } finally {

                loadingElement.classList.add('hidden');

            }

        }


        /**

         * Khởi tạo tạo ảnh khi trang được tải

         */

        window.onload = function () {

            // 1. Ảnh Chủ quán cầm tablet quản lý

            const ownerPrompt = "A smiling young coffee shop owner, wearing a modern apron, efficiently managing orders and inventory on a sleek, large tablet in a bright, minimalist cafe. Professional, vivid, digital art style. Full body shot.";

            generateImage(ownerPrompt, 'ownerImage', 'ownerLoading');


            // 2. Ảnh Ly cà phê (Điểm nhấn)

            const coffeePrompt = "A close-up of a beautifully designed latte art coffee cup on a wooden counter with a soft, blurred background of a cozy coffee shop. Warm lighting, high detail, photo-realistic.";

            generateImage(coffeePrompt, 'coffeeImage', 'coffeeLoading');

            

            // Khởi tạo Lucide Icons lần nữa

            lucide.createIcons();

        }
